# Calculate Shannon entropy for each match
# H_i = -sum(p_k * log_3(p_k)) for k in {w, d, l}
calculate_entropy <- function(p_win, p_draw, p_loss) {
  # Handle zero probabilities to avoid log(0)
  probs <- c(p_win, p_draw, p_loss)
  probs <- probs[probs > 0]  # Remove zero probabilities
  if (length(probs) == 0) return(0)
  -sum(probs * log(probs, base = 3))
}

# Vectorized entropy calculation
calc_entropy_vectorized <- function(p_win, p_draw, p_loss) {
  entropy <- rep(0, length(p_win))
  
  # For home team matches
  h_mask <- !is.na(p_win) & !is.na(p_draw) & !is.na(p_loss)
  pw <- p_win[h_mask]
  pd <- p_draw[h_mask]
  pl <- p_loss[h_mask]
  
  ent <- rep(0, sum(h_mask))
  if (pw > 0) ent <- ent - pw * log(pw, base = 3)
  if (pd > 0) ent <- ent - pd * log(pd, base = 3)
  if (pl > 0) ent <- ent - pl * log(pl, base = 3)
  
  entropy[h_mask] <- ent
  return(entropy)
}
