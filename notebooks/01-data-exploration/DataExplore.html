Team level dataset

home_team <- games %>%
  transmute(
    Date,
    Season,
    Team = HomeTeam,
    Opponent = AwayTeam,
    GoalsFor = FTHG,
    GoalsAgainst = FTAG,
    Home = 1,
    Odds = B365H,
    OddsDraw = B365D,
    Result = case_when(
      FTR == "H" ~ "W",
      FTR == "D" ~ "D",
      FTR == "A" ~ "L"
    )
  )

away_team <- games %>%
  transmute(
    Date,

      FTR == "A" ~ "W",
