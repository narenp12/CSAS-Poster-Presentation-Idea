# Apply power transformation (Clarke, 2017) to convert raw odds to calibrated probabilities
odds_input <- games %>% select(B365H, B365D, B365A)
transformed <- implied_probabilities(odds_input, method = "power")
prob_df <- data.frame(transformed$probabilities) %>%
  rename(p_home = B365H, p_draw = B365D, p_away = B365A)

# Compute Shannon entropy from power-transformed probabilities
# H_i = -sum(p_k * log_3(p_k)) for k in {home, draw, away}
games_processed <- games %>%
  bind_cols(prob_df) %>%
  mutate(
    entropy = -(p_home * log(p_home, base = 3) +
                  p_draw * log(p_draw, base = 3) +
                  p_away * log(p_away, base = 3)),
    Date = as.Date(Date)
  )

Compute Season Standings and Entropy Features

compute_season_standings <- function(season_num) {
  seasonal_file <- season_files[str_detect(season_files, sprintf("%04d", season_num))]
  if (length(seasonal_file) == 0) return(NULL)
  if (!file.exists(seasonal_file)) return(NULL)

  seasonal <- read_csv(seasonal_file, show_col_types = FALSE)

  home <- seasonal %>%
    select(team = HomeTeam, FTHG, FTAG, FTR) %>%
    mutate(
      points = case_when(FTR == "H" ~ 3, FTR == "D" ~ 1, TRUE ~ 0),
      gd = FTHG - FTAG
    )

  away <- seasonal %>%
    select(team = AwayTeam, FTHG, FTAG, FTR) %>%
    mutate(
      points = case_when(FTR == "A" ~ 3, FTR == "D" ~ 1, TRUE ~ 0),
      gd = FTAG - FTHG
    )

  bind_rows(home, away) %>%
    group_by(team) %>%
    summarise(
      final_points = sum(points),
      goal_difference = sum(gd),
      .groups = "drop"
    ) %>%
    arrange(desc(final_points), desc(goal_difference)) %>%
    mutate(
      final_position = row_number(),
      relegated = final_position >= 18,
      season = season_num
    )
}

Bayesian Hierarchical Logistic Regression

# Model: Relegation ~ mean_entropy + var_entropy + (1 | season)
# Varying intercepts by season account for competitive balance differences

model_formula <- bf(relegated ~ mean_entropy_z + var_entropy_z + (1 | season))

model_priors <- c(
  prior(normal(0, 1.5), class = "b"),      # Population-level effects
  prior(normal(0, 1), class = "sd"),       # Season-level variation
  prior(normal(0, 2), class = "Intercept") # Base relegation rate
)

set.seed(42)
fit_relegation <- brm(
  model_formula,
  data = model_data,
  family = bernoulli(),
  prior = model_priors,
  chains = 4,
  iter = 4000,
  warmup = 1000,
  control = list(adapt_delta = 0.95),
  seed = 42,
  file = "fit_bayesian_entropy_relegation"
)

compare_models <- function(fits, names) {
  results <- map2_dfr(fits, names, function(fit, nm) {
    post <- as_draws_df(fit)
    tibble(
      model = nm,
      parameter = c("Mean Entropy", "Entropy Variance", "Intercept"),
      median = c(
        median(post$b_mean_entropy_z),
        median(post$b_var_entropy_z),
        median(post$b_Intercept)
      ),
      lower = c(
        quantile(post$b_mean_entropy_z, 0.025),
        quantile(post$b_var_entropy_z, 0.025),
        quantile(post$b_Intercept, 0.025)
      ),
      upper = c(
        quantile(post$b_mean_entropy_z, 0.975),
        quantile(post$b_var_entropy_z, 0.975),
        quantile(post$b_Intercept, 0.975)
      )
    )
  })
  return(results)
}
