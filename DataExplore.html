  map(~ .x %>%
    select(Date, HomeTeam, AwayTeam, FTHG, FTAG, FTR, B365H, B365D, B365A) %>%
    mutate(Date = dmy(Date)) %>%
    mutate(Season = year(max(Date)))) %>%
  bind_rows()

Team level dataset

home_team <- games %>%
  transmute(
    Date,
    Season,
    Team = HomeTeam,
    Opponent = AwayTeam,
    GoalsFor = FTHG,
    GoalsAgainst = FTAG,
    Home = 1,
    Odds = B365H,
    OddsDraw = B365D,
    Result = case_when(
      FTR == "H" ~ "W",
      FTR == "D" ~ "D",
      FTR == "A" ~ "L"
    )
  )

away_team <- games %>%
  transmute(
    Date,
    Season,
    Team = AwayTeam,
    Opponent = HomeTeam,
    GoalsFor = FTAG,
    GoalsAgainst = FTHG,
    Home = 0,
    Odds = B365A,
    OddsDraw = B365D,
    Result = case_when(
      FTR == "A" ~ "W",
      FTR == "D" ~ "D",
      FTR == "H" ~ "L"
    )
  )
